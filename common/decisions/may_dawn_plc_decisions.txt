# Decyzja 1: Śląsk i Prusy
plc_claim_western_lands = {
    is_shown = {
		exists = c:PLC
		c:PLC ?= THIS
	}

    possible = {
		c:PLC ?= THIS
		has_technology_researched = nationalism
		hidden_trigger = {
			NOT = { has_variable = plc_western_claims_done }
		}
	}

    possible = {
        country_rank >= rank_value:major_power # Musisz być przynajmniej mocarstwem regionalnym
    }

    when_taken = {
    	set_variable = {
			name = plc_western_claims_done
			value = yes
		}
        # Górny i Dolny Śląsk
        s:STATE_LOWER_SILESIA = { add_claim = c:PLC }
        s:STATE_UPPER_SILESIA = { add_claim = c:PLC }
        # Prusy Wschodnie i Zachodnie
        s:STATE_EAST_PRUSSIA = { add_claim = c:PLC }
        change_infamy = 15

    }

    ai_chance = { value = 1 }
}
