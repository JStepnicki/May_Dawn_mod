# Decyzja 1: Śląsk i Prusy
plc_claim_western_lands = {
    is_shown = {
        # Pokazuj dla PLC, PEC (Imperium) lub PLR (Rzeczpospolita Wielu Narodów)
        OR = {
            c:PLC ?= THIS
            c:PEC ?= THIS
            c:PLR ?= THIS
        }
        NOT = { has_variable = plc_western_claims_done }
    }

    possible = {
        OR = {
            c:PLC ?= THIS
            c:PEC ?= THIS
            c:PLR ?= THIS
        }
        has_technology_researched = nationalism
        country_rank >= rank_value:major_power
        hidden_trigger = {
            NOT = { has_variable = plc_western_claims_done }
        }
    }

    when_taken = {
        set_variable = {
            name = plc_western_claims_done
            value = yes
        }
        # Roszczenia przypisują się do aktualnego tagu (THIS)
        s:STATE_LOWER_SILESIA = { add_claim = ROOT }
        s:STATE_UPPER_SILESIA = { add_claim = ROOT }
        s:STATE_EAST_PRUSSIA = { add_claim = ROOT }
        s:STATE_WEST_PRUSSIA = { add_claim = ROOT }

        change_infamy = 15
    }

    ai_chance = { value = 1 }
}
