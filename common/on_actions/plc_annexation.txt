# --- PLC ANNEXATION ON ACTIONS ---
# This file ensures that the last "stub" provinces of the occupiers
# are transferred to PLC/COU on day 0.

on_game_started = {
    on_actions = {
        plc_history_setup_on_action
    }
}

plc_history_setup_on_action = {
    effect = {
        # --- MAZOVIA (Stealing the last province from Russia) ---
        s:STATE_MAZOVIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x009F60 } # The "capitol" province of the state
            }
        }

        # --- GREATER POLAND ---
        s:STATE_GREATER_POLAND = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x0E40DA }
            }
        }

        # --- LESSER POLAND ---
        s:STATE_LESSER_POLAND = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x40DF20 }
            }
        }

        # --- POSEN (Stealing the last part from Prussia) ---
        s:STATE_POSEN = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x1FA1E1 }
            }
        }

        # --- WEST PRUSSIA (Gdańsk/Danzig area) ---
        s:STATE_WEST_PRUSSIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x178C99 }
            }
        }

        # --- WEST GALICIA ---
        s:STATE_WEST_GALICIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x90E060 }
            }
        }

        # --- EAST GALICIA (Lwów area) ---
        s:STATE_EAST_GALICIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { xA2805F }
            }
        }

        # --- COURLAND (Transferring to Courland tag) ---
        s:STATE_COURLAND = {
            set_owner_of_provinces = {
                country = c:COU
                provinces = { x01E021 }
            }
        }

        # --- LITHUANIA / KAUNAS ---
        s:STATE_KAUNAS = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x00E0A0 }
            }
        }

        # --- VILNIUS ---
        s:STATE_VILNIUS = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x0161A0 }
            }
        }

        # --- VOLHYNIA ---
        s:STATE_VOLHYNIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x0EB3C7 }
            }
        }

        # --- BREST ---
        s:STATE_BREST = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x157352 }
            }
        }

        # NOTE: For Split States like KIEV or MINSK, do NOT add them here
        # unless you want to annex the Russian parts as well!
    }
}
