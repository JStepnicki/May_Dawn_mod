# --- PLC ANNEXATION ON ACTIONS ---
# This file ensures that the last "stub" provinces of the occupiers
# are transferred to PLC/COU on day 0.

on_game_started = {
    on_actions = {
        plc_history_setup_on_action
    }
}

plc_history_setup_on_action = {
    effect = {
        # --- MAZOVIA (Stealing the last province from Russia) ---
        s:STATE_MAZOVIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x43B604 } # The "capitol" province of the state
            }
        }

        # --- GREATER POLAND ---
        s:STATE_GREATER_POLAND = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { xB47E64 }
            }
        }

        # --- LESSER POLAND ---
        s:STATE_LESSER_POLAND = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x49500A }
            }
        }

        # --- POSEN (Stealing the last part from Prussia) ---
        s:STATE_POSEN = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x6D42A4 }
            }
        }

        # --- WEST PRUSSIA (Gdańsk/Danzig area) ---
        s:STATE_WEST_PRUSSIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x51486A }
            }
        }

        # --- WEST GALICIA ---
        s:STATE_WEST_GALICIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x778604 }
            }
        }

        # --- EAST GALICIA (Lwów area) ---
        s:STATE_EAST_GALICIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { xA2C490 }
            }
        }

        # --- COURLAND (Transferring to Courland tag) ---
        s:STATE_COURLAND = {
            set_owner_of_provinces = {
                country = c:COU
                provinces = { x5C2A40 }
            }
        }

        # --- LITHUANIA / KAUNAS ---
        s:STATE_KAUNAS = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x6626DA }
            }
        }

        # --- VILNIUS ---
        s:STATE_VILNIUS = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x6C2754 }
            }
        }

        # --- VOLHYNIA ---
        s:STATE_VOLHYNIA = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { xA65034 }
            }
        }

        # --- BREST ---
        s:STATE_BREST = {
            set_owner_of_provinces = {
                country = c:PLC
                provinces = { x89512A }
            }
        }

        # NOTE: For Split States like KIEV or MINSK, do NOT add them here
        # unless you want to annex the Russian parts as well!
    }
}
